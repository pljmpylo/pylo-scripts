      /* ═══════════ SWIPE NAVIGATION ═══════════ */
      (function() {
      const TABS = ['home', 'sortie', 'program', 'progres'];
      const MIN_DIST = 70;
      const MAX_VERT = 40; // max vertical movement allowed
      let startX = 0, startY = 0;
      let isSwiping = false;
      document.addEventListener('touchstart', e => {
      startX = e.touches[0].clientX;
      startY = e.touches[0].clientY;
      isSwiping = false;
      }, {passive: true});
      document.addEventListener('touchmove', e => {
      const dx = Math.abs(e.touches[0].clientX - startX);
      const dy = Math.abs(e.touches[0].clientY - startY);
      // Mark as horizontal swipe early if clearly horizontal
      if (dx > 15 && dx > dy * 2) isSwiping = true;
      // Cancel if vertical
      if (dy > MAX_VERT && dy > dx) isSwiping = false;
      }, {passive: true});
      document.addEventListener('touchend', e => {
      if (!isSwiping) return;
      const dx = e.changedTouches[0].clientX - startX;
      if (Math.abs(dx) < MIN_DIST) return;
      // Don't swipe if a panel or modal is open
      if (document.getElementById('view-settings').style.visibility === 'visible') return;
      if (document.getElementById('view-guide').style.visibility === 'visible') return;
      if (document.getElementById('view-install').style.visibility === 'visible') return;
      if (document.getElementById('view-calendar-guide').style.visibility === 'visible') return;
      if (document.getElementById('sfb-modal').classList.contains('open')) return;
      if (document.getElementById('wci-modal').classList.contains('open')) return;
      if (document.getElementById('view-download').style.display === 'block') return;
      const idx = TABS.indexOf(currentTab);
      if (idx === -1) return;
      if (dx < 0 && idx < TABS.length - 1) switchTab(TABS[idx + 1]);
      if (dx > 0 && idx > 0) switchTab(TABS[idx - 1]);
      isSwiping = false;
      }, {passive: true});
      })();
